#include <iostream>
#include <fstream>
#include <string>
using namespace std;

const int SIZE = 8;

void readCities(string cities[], int population[]) {
    ifstream inFile("cities.txt");
    for (int i = 0; i < SIZE; i++) {
        inFile >> cities[i] >> population[i];
    }
    inFile.close();
}

int findCity(const string cities[], const string &search) {
    for (int i = 0; i < SIZE; i++) {
        if (cities[i] == search)
            return i;
    }
    return -1;
}

int main() {
    string cities[SIZE];
    int population[SIZE];
    string cityName;

    readCities(cities, population);

    cout << "Enter city name, Ctrl+Z to stop: ";
    while (cin >> cityName) {
        int index = findCity(cities, cityName);
        if (index != -1)
            cout << cityName << " has a population of " << population[index] << endl;
        else
            cout << cityName << " not found" << endl;

        cout << "\nEnter city name, Ctrl+Z to stop: ";
    }

    cout << "\nGoodbye. Have a nice day.\n";
    return 0;
}
